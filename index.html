    function calculerDevis() {

      const type = document.getElementById('typeLogement').value;

      const etage = parseInt(document.getElementById('etage').value);

      const ascenseur = document.getElementById('ascenseur').checked;

      let total = 0;

      if(document.getElementById('debarassage').checked) total += prix[type].debarassage;

      if(document.getElementById('demenagement').checked) total += prix[type].demenagement;

      if(document.getElementById('nettoyageN1').checked) total += prix[type].nettoyageN1;

      if(document.getElementById('nettoyageN2').checked) total += prix[type].nettoyageN2;

      if(document.getElementById('nettoyageN3').checked) total += prix[type].nettoyageN3;

      if(document.getElementById('rampants').checked) total += prix[type].rampants;

      if(document.getElementById('epiStandard').checked) total += prix[type].epiStandard;

      if(document.getElementById('epiSpeciaux').checked) total += prix[type].epiSpeciaux;

      if(document.getElementById('epiDemenagement').checked) total += prix[type].epiDemenagement;

      // Frais annexes

      total += prix[type].fraisAnnexes;

      // Calcul étage

      if(!ascenseur && etage > 0) {

        if(etage === 1) total += prix[type].etagesN1 + prix[type].etagesN2;

        else if(etage === 2) total += prix[type].etages2N1 + prix[type].etages2N2;

        else if(etage === 3) total += prix[type].etages3N1 + prix[type].etages3N2;

        else if(etage === 4) total += prix[type].etages4N1 + prix[type].etages4N2;

      }

      document.getElementById('result').innerText = "Devis estimatif : " + total + " €";

    }
